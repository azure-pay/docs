This guide creates one checkout session and mounts the VoucherPay widget.

## 1. Install packages

```bash
npm install @voucherpay/server @azure-pay/checkout-widget
```

The current package names are:

| Package | Runs where | Purpose |
| --- | --- | --- |
| `@voucherpay/server` | Backend only | Creates sessions and verifies webhooks. |
| `@azure-pay/checkout-widget` | Browser | Mounts VoucherPay checkout. |

## 2. Configure environment variables

```bash
VOUCHERPAY_API_BASE_URL=https://api-demo-02f5.up.railway.app
VOUCHERPAY_SECRET_KEY=vpay_test_...
VOUCHERPAY_WEBHOOK_SECRET=whsec_...
NEXT_PUBLIC_VOUCHERPAY_API_BASE_URL=https://api-demo-02f5.up.railway.app
```

## 3. Create a backend route

This example is a Next.js route handler. The same pattern applies to Express, Rails, Laravel, Django, or any backend.

```ts
// app/api/voucherpay/checkout/route.ts
import { VoucherPay } from "@voucherpay/server";

const voucherpay = new VoucherPay(process.env.VOUCHERPAY_SECRET_KEY!, {
  apiBaseUrl: process.env.VOUCHERPAY_API_BASE_URL,
});

export async function POST(request: Request) {
  const body = (await request.json()) as {
    orderId: string;
    customerEmail?: string;
  };

  // Load the order from your database. Do not trust client-supplied totals.
  const order = {
    id: body.orderId,
    amountCents: 9900,
    currency: "EUR" as const,
    description: "Order " + body.orderId,
  };

  const session = await voucherpay.sessions.create(
    {
      externalReference: order.id,
      amountCents: order.amountCents,
      currency: order.currency,
      customerEmail: body.customerEmail,
      productTitle: order.description,
      successRedirect: "https://store.example.com/thanks",
      failureRedirect: "https://store.example.com/checkout?failed=1",
      cancelRedirect: "https://store.example.com/checkout?cancelled=1",
      inProgressRedirect: "https://store.example.com/checkout?pending=1",
      metadata: {
        orderId: order.id,
      },
    },
    {
      idempotencyKey: order.id,
    },
  );

  return Response.json({
    sessionId: session.id,
    clientSecret: session.clientSecret,
  });
}
```

Important:

- The amount and currency must come from your server-side order record.
- Use a stable `Idempotency-Key`, usually your immutable order ID.
- Checkout creation currently accepts `EUR` only.

## 4. Mount the widget

```tsx
"use client";

import { useEffect, useRef, useState } from "react";
import { mount, type WidgetHandle } from "@azure-pay/checkout-widget";

export function VoucherPayCheckout({ orderId }: { orderId: string }) {
  const containerRef = useRef<HTMLDivElement | null>(null);
  const handleRef = useRef<WidgetHandle | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;

    async function start() {
      const response = await fetch("/api/voucherpay/checkout", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ orderId }),
      });

      const data = (await response.json()) as {
        sessionId: string;
        clientSecret: string;
      };

      if (cancelled || !containerRef.current) return;

      handleRef.current = mount({
        target: containerRef.current,
        sessionId: data.sessionId,
        clientSecret: data.clientSecret,
        apiBaseUrl: process.env.NEXT_PUBLIC_VOUCHERPAY_API_BASE_URL,
        onSuccess: ({ paymentSessionId, voucherReference }) => {
          console.log("VoucherPay captured", paymentSessionId, voucherReference);
        },
        onError: ({ code, message }) => {
          setError(`${code}: ${message}`);
        },
      });
    }

    void start().catch((err) => {
      setError(err instanceof Error ? err.message : "Checkout failed");
    });

    return () => {
      cancelled = true;
      handleRef.current?.destroy();
    };
  }, [orderId]);

  if (error) return <p>{error}</p>;

  return <div ref={containerRef} />;
}
```

## 5. Fulfill with webhooks

Use the success callback to update the page, but use signed webhooks to mark the order paid in your backend.
